/**
 * Golden-fixture generator for issue #39 (linky-next): pins the PoC's
 * LNURL-pay WIRE behavior — the exact HTTP request URLs the PoC issues for
 * the pay-metadata load and the invoice callback (amount param, LUD-12
 * comment handling incl. the silent no-comment retry), plus the min/max sat
 * preview rounding.
 *
 * Run from linky-poc/apps/web-app with bun, BEFORE asserting anything with
 * linky-next code (anything else would be circular):
 *
 *   bun __gen_lnurlpay_callback_fixture.ts > /tmp/lnurlPayCallback.golden.json
 *
 * The global fetch is stubbed: every request URL is recorded and answered
 * with the scripted JSON for the scenario, so the fixture captures the PoC's
 * own URL construction (URL.searchParams) and fallback logic verbatim.
 */
import {
  fetchLnurlInvoiceForTarget,
  fetchLnurlPayPreview,
} from "./src/lnurlPay";

interface RecordedCall {
  url: string;
  status: number;
}

const PAY_REQUEST_BASE = {
  tag: "payRequest",
  callback: "https://fixture.test/lnurlp/callback?session=abc",
  minSendable: 1500,
  maxSendable: 250500,
  metadata: JSON.stringify([
    ["text/plain", "Fixture pay"],
    ["image/png;base64", "aWNvbg=="],
  ]),
};

type Responder = (url: string, callIndex: number) => { status: number; body: unknown };

const runScenario = async (
  respond: Responder,
  action: () => Promise<unknown>,
): Promise<{ requests: RecordedCall[]; result: unknown; error: string | null }> => {
  const requests: RecordedCall[] = [];
  let call = 0;
  (globalThis as { fetch: unknown }).fetch = async (input: unknown) => {
    const url = String(input);
    const { status, body } = respond(url, call);
    call += 1;
    requests.push({ url, status });
    return new Response(JSON.stringify(body), {
      status,
      headers: { "content-type": "application/json" },
    });
  };
  try {
    const result = await action();
    return { requests, result, error: null };
  } catch (error) {
    return { requests, result: null, error: String(error instanceof Error ? error.message : error) };
  }
};

const invoiceResponse = { pr: "fixtureinvoice", routes: [] };

const main = async () => {
  // A. Preview load for a lightning address: pins the well-known URL and the
  //    conservative sat rounding (ceil min, floor max) + commentAllowed=0.
  const preview = await runScenario(
    () => ({ status: 200, body: PAY_REQUEST_BASE }),
    () => fetchLnurlPayPreview("alice@fixture.test"),
  );

  // B. Invoice fetch without a comment: amount param appended to a callback
  //    that already carries a query string.
  const invoiceNoComment = await runScenario(
    (url) =>
      url.includes("/callback")
        ? { status: 200, body: invoiceResponse }
        : { status: 200, body: PAY_REQUEST_BASE },
    () => fetchLnurlInvoiceForTarget("alice@fixture.test", 21),
  );

  // C. Comment with an ADVERTISED commentAllowed=12: truncated, single call.
  const advertised = { ...PAY_REQUEST_BASE, commentAllowed: 12 };
  const invoiceAdvertisedComment = await runScenario(
    (url) =>
      url.includes("/callback")
        ? { status: 200, body: invoiceResponse }
        : { status: 200, body: advertised },
    () => fetchLnurlInvoiceForTarget("alice@fixture.test", 21, "Hello from Linky"),
  );

  // D. Comment with NO advertised support: tried with a ≤140-char comment
  //    first; the 500 makes the PoC silently retry without it.
  const invoiceFallbackComment = await runScenario(
    (url) => {
      if (!url.includes("/callback")) return { status: 200, body: PAY_REQUEST_BASE };
      return url.includes("comment=")
        ? { status: 500, body: { status: "ERROR", reason: "no comments" } }
        : { status: 200, body: invoiceResponse };
    },
    () => fetchLnurlInvoiceForTarget("alice@fixture.test", 21, "Hello from Linky"),
  );

  const fixture = {
    generatedFrom:
      "linky-poc apps/web-app/src/lnurlPay.ts (fetchLnurlPayPreview / fetchLnurlInvoiceForTarget), bun, fetch stubbed; generator: __gen_lnurlpay_callback_fixture.ts",
    payRequest: PAY_REQUEST_BASE,
    scenarios: {
      preview,
      invoiceNoComment,
      invoiceAdvertisedComment,
      invoiceFallbackComment,
    },
  };
  console.log(JSON.stringify(fixture, null, 2));
};

await main();
