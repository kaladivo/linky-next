// --- NIP-98 hosted mint sync request (PoC makeNip98AuthHeader + updateNpubCashMint) ---
const NIP98_PRIV_HEX = bytesToHex(sha256(new TextEncoder().encode("linky-#41-golden-nip98")));
const NIP98_PRIV = (() => {
  const bytes = new Uint8Array(32);
  for (let i = 0; i < 32; i += 1) {
    bytes[i] = Number.parseInt(NIP98_PRIV_HEX.slice(i * 2, i * 2 + 2), 16);
  }
  return bytes;
})();

interface Nip98Case {
  name: string;
  baseUrl: string;
  mintUrl: string;
  nowSec: number;
}

const NIP98_CASES: Nip98Case[] = [

const makeNip98Fixture = async ({ name, baseUrl, mintUrl, nowSec }: Nip98Case) => {
  // Exactly what useNpubCashMintSelection.updateNpubCashMint does:
  const cleaned = normalizeMintUrl(mintUrl);
  const url = `${baseUrl}/api/v1/info/mint`;
  const payload = { mintUrl: cleaned };

  // nip98.getToken uses `new Date().getTime()` — freeze it for determinism.
  const RealDate = Date;
  const fixedMs = nowSec * 1000;
  // @ts-expect-error deliberate global override
  globalThis.Date = class extends RealDate {
    constructor() {
      super(fixedMs);
    }
    static now() {
      return fixedMs;
    }
  };
  let token: string;
  try {
    token = await nip98.getToken(
      url,
      "PUT",
      async (event) => finalizeEvent(event as never, NIP98_PRIV),
      true,
      payload,
    );
  } finally {
    globalThis.Date = RealDate;
  }

  const eventJson = new TextDecoder().decode(
    Uint8Array.from(atob(token.slice("Nostr ".length)), (c) => c.charCodeAt(0)),
  );
  const eventJsonSigMasked = eventJson.replace(/"sig":"[0-9a-f]{128}"/, '"sig":"SIG"');

  return {
    name,
    baseUrl,
    inputMintUrl: mintUrl,
    nowSec,
    request: {
      url,
      method: "PUT",
      contentType: "application/json",
      body: JSON.stringify(payload),
    },
    authorizationScheme: "Nostr ",
    eventJsonSigMasked,
    payloadHashHex: bytesToHex(sha256(new TextEncoder().encode(JSON.stringify(payload)))),
  };
};
